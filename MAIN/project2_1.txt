START 200
EXTERN EXTERN_VAR1
EXTERN EXTERN_VAR2
EXTERN EXTERN_VAR3
ENTRY FOUND
ENTRY NOT_FOUND
MOVER AREG, ARRAY
MOVER CREG, LENGTH
MOVER DREG, TARGET
SEARCH_LOOP: COMP AREG, DREG
JZ SEARCH_FOUND:
INCR AREG
DECR CREG
JNZ SEARCH_LOOP:
MOVER AREG, NOT_FOUND
JMP SEARCH_END:
SEARCH_FOUND: MOVER AREG, FOUND
SEARCH_END: STOP
ARRAY DS 5 
TARGET DC 1
LENGTH DC 1
FOUND DC 1
NOT_FOUND DC 1
END
