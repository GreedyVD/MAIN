START 210
EXTERN FOUND
EXTERN NOT_FOUND
EXTERN EXTERN_VAR4
ENTRY NEW_FOUND
ENTRY NEW_TARGET
MOVER AREG, EXTERN_VAR4
COMP AREG, FOUND
JZ NEW_FOUND_LABEL
MOVER BREG, NOT_FOUND
JMP END_LABEL
NEW_FOUND_LABEL: MOVER AREG, NEW_FOUND
END_LABEL: STOP
EXTERN_VAR1 DC 1
EXTERN_VAR2 DC 1
EXTERN_VAR5 DC 1
NEW_FOUND DC 1
NEW_TARGET DS 1
END
